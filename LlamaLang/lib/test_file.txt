int gcd(int a, int b):
	if b == 0:
		return a
	else:
		return gcd(b, a % b)

int binarySearch(int[ ] arr, int left, int right, int x):
	while left <= right:
		
		int mid = left + (right - left) // 2
		
		if arr[mid] ==  x:
			return mid
		else if arr[mid] < x:
			left = mid + 1
		else:
			right = mid - 1
	return -1
